int fib(int n)
{
    if (n > 1)
        return fib(n - 1) + fib(n - 2);
    else
        return 1;
}

float pi(int n)
{
    float sum = 0;
    float sig = 1.0;
    for (int i = 1; i <=n; i += 2)
    {
        sum += sig;   
        sig = -sig;
    }  
    return sum;
}

{
    int  fib_v;
    float pi_v;
}v;

    print(v.pi_v + v.fib_v + 1.0);

float pi(int n)
{
    float sum = 0;
    float sig = 1.0;
    for (int i = 1; i <=n; i += 2)
    {
        sum += sig/i;   
        sig = 0-sig;
    }  
    return sum;
}

       print(iarr[i]);
       print(farr[i]);
    }
    print(iarr[3]);
    pi(iarr[2]);
    return 0;
}
